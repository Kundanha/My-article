from collections import Counter

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        count = Counter(nums)
        
        # Min heap of size k
        minHeap = []
        
        for num, freq in count.items():
            heapq.heappush(minHeap, (freq, num))
            
            # Keep only k elements
            if len(minHeap) > k:

        # Extract elements from heap

from collections import Counter

class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        count = Counter(nums)
        n = len(nums)
        
        # Create buckets: bucket[i] = elements with frequency i
        buckets = [[] for _ in range(n + 1)]
        
        for num, freq in count.items():
            buckets[freq].append(num)
        
        # Collect k elements from highest frequency buckets
        result = []
        for freq in range(n, 0, -1):  # n to 1 (reverse)
            for num in buckets[freq]:
                result.append(num)
                if len(result) == k:
                    return result
        
        return result

            return ""

        res = ""
        
        # Collect all lengths
        for s in strs:
            sizes.append(len(s))
        
        # Write lengths with comma separator
        for sz in sizes:
            res += str(sz) + ','
        
        res += '#'  # Boundary marker
        
        # Append all strings
        for s in strs:
            res += s
        
        return res

    def decode(self, s: str) -> List[str]:
        if not s:
            return []
        
        sizes, res, i = [], [], 0
        
        # Extract all sizes until '#'
        while s[i] != '#':
            cur = ""
            while s[i] != ',':
                cur += s[i]
                i += 1
            sizes.append(int(cur))
            i += 1  # Skip comma
        
        i += 1  # Skip '#'
        
        # Extract strings based on sizes
        for sz in sizes:
            res.append(s[i:i + sz])
            i += sz
        
        return res

    def decode(self, s: str) -> List[str]:

        i = 0
        
        while i < len(s):
            # Find the '#' separator
            j = i
            while s[j] != '#':
                j += 1
            
            # Extract length
            length = int(s[i:j])
            
            # Move past '#'
            i = j + 1
            
            # Extract the string of 'length' characters
            j = i + length
            res.append(s[i:j])
            
            # Move to next segment
            i = j
        
        return res

class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        res = []
        nums.sort()
        
        for i, a in enumerate(nums):
            # Early termination: if a > 0, no way to get sum = 0
            if a > 0:
                break
            
            # Skip duplicate values for first element
            if i > 0 and a == nums[i - 1]:
                continue
            
            # Two pointers for remaining two elements
            l, r = i + 1, len(nums) - 1
            
            while l < r:
                threeSum = a + nums[l] + nums[r]
                
                if threeSum > 0:
                    r -= 1  # Sum too big, decrease
                elif threeSum < 0:
                    l += 1  # Sum too small, increase
                else:
                    # Found a valid triplet!
                    res.append([a, nums[l], nums[r]])
                    l += 1
                    r -= 1
                    
                    # Skip duplicates for left pointer
                    while nums[l] == nums[l - 1] and l < r:
                        l += 1
        
        return res

# ============================================
# ðŸ” BINARY SEARCH: Find Minimum in Rotated Array
# ============================================
# Key Insight: Rotated array = 2 sorted halves!
# Ek half HIGHER values wala, ek LOWER values wala
# Minimum = Lower half ka first element ðŸŽ¯

def findMin(nums):
    """
    Binary Search approach jo sorted property use karta hai
    Har step mein half array eliminate karte hain!
    
    Time: O(log n) â€” kyunki half-half karte hain
    Space: O(1) â€” sirf pointers use kar rahe
    """
    
    # Step 1: Initialize karo pointers aur result
    res = nums[0]              # Pehle element se start karo
    left = 0                   # Left pointer
    right = len(nums) - 1      # Right pointer
    
    # Step 2: Binary search loop
    while left <= right:
        
        # ðŸŽ¯ OPTIMIZATION: Agar current window already sorted hai
        # toh minimum left pe hi hai â€” no need to search further!
        if nums[left] < nums[right]:
            res = min(res, nums[left])
            break   # Early exit! ðŸš€
        
        # Middle element calculate karo
        mid = (left + right) // 2
        
        # Current mid ko consider karo as potential answer
        res = min(res, nums[mid])
        
        # ðŸ¤” DECIDE: Konsi side jaana hai?
        # Agar nums[mid] >= nums[left] â†’ Left half sorted hai
        # Matlab rotation RIGHT side mein hai!
        if nums[mid] >= nums[left]:
            # Left side sorted (higher values) â†’ search RIGHT
            left = mid + 1
        else:
            # Right side sorted (lower values) â†’ search LEFT
            right = mid - 1
    
    return res

# ============================================
# âœ¨ LOWER BOUND: Cleanest Binary Search Solution
# ============================================
# Sabse elegant approach! Sirf right se compare karo
# Loop end hone pe left automatically minimum pe hoga ðŸŽ¯

def findMin(nums):
    """
    Lower Bound technique â€” sabse clean implementation!
    
    ðŸ’¡ Key Insight:
    - nums[mid] < nums[right] â†’ mid "lower half" mein hai
      â†’ minimum LEFT mein hai (mid included)
    - nums[mid] > nums[right] â†’ mid "higher half" mein hai  
      â†’ minimum RIGHT mein hai (mid excluded)
    
    Time: O(log n)
    Space: O(1)
    """
    
    # Two pointers initialize karo
    left = 0
    right = len(nums) - 1
    
    # Jab tak pointers converge nahi karte
    while left < right:   # Note: < NOT <=
        
        # ðŸ›¡ï¸ Overflow-safe mid calculation
        # (left + right) // 2 se overflow ho sakta hai large arrays mein
        mid = left + (right - left) // 2
        
        # ðŸ”‘ THE KEY COMPARISON: mid vs right
        if nums[mid] < nums[right]:
            # âœ… Mid LOWER half mein hai (smaller values)
            # Minimum left mein ho sakta hai, mid pe bhi ho sakta hai!
            right = mid     # ðŸ‘ˆ mid include karo!
            
        else:  # nums[mid] > nums[right]
            # âœ… Mid HIGHER half mein hai (larger values)
            # Minimum definitely right mein hai, mid pe nahi
            left = mid + 1  # ðŸ‘ˆ mid exclude karo!
    
    # ðŸŽ¯ Loop khatam hone pe left = right = minimum index
    return nums[left]

# ============================================
# ðŸ” TWO PASS: Pivot Dhundho + Binary Search
# ============================================
# Pass 1: Pehle minimum element (pivot) dhundho
# Pass 2: Phir correct half mein binary search karo
# 
# ðŸ’¡ Rotated array = 2 sorted halves joined together!

def search(nums, target):
    """
    Two Pass Binary Search
    
    ðŸŽ¯ Strategy:
    1. Find pivot (index of minimum element)
    2. Decide which half to search
    3. Run standard binary search on that half
    
    Time: O(log n) â€” 2 binary searches
    Space: O(1)
    """
    n = len(nums)
    
    # ========================================
    # ðŸ“ PASS 1: Find Pivot (Minimum Element)
    # ========================================
    # Pivot = jahan se rotation shuru hui
    # Iske left mein bade elements, right mein chhote
    
    left, right = 0, n - 1
    
    while left < right:
        mid = (left + right) // 2
        
        # Agar mid > right â†’ minimum RIGHT mein hai
        if nums[mid] > nums[right]:
            left = mid + 1
        else:
            # Minimum LEFT mein hai (including mid)
            right = mid
    
    # ðŸŽ¯ Ab left = right = pivot index
    pivot = left
    
    # ========================================
    # ðŸ“ PASS 2: Decide Which Half to Search
    # ========================================
    # Reset pointers
    left, right = 0, n - 1
    
    # Target kis half mein hai?
    # Right half: [pivot, n-1] mein smallest se largest tak
    if target >= nums[pivot] and target <= nums[right]:
        # âœ… Target RIGHT half mein hai
        left = pivot
    else:
        # âœ… Target LEFT half mein hai
        right = pivot - 1
    
    # ========================================
    # ðŸ“ Standard Binary Search
    # ========================================
    while left <= right:
        mid = (left + right) // 2
        
        if nums[mid] == target:
            # ðŸŽ‰ Mil gaya!
            return mid
        elif nums[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    
    # âŒ Nahi mila
    return -1
